\subsubsection{クラス VisualStudio}
\label{subsubsec:VisualStudio}

\bigskip
\begin{Brief}[b]
	Visual studioをコマンドラインから使用するためのラッパクラス。
	Visual Studio 2015以降に対応。
	ただし、Visual Stduio 2017以降については未確認である
	(tool setの表記法、\tt{devenv}の格納ディレクトリ等が変更されているかもしれない）。
\end{Brief}

\bigskip
\begin{Initializer}{
	obj = VisualStudio(toolset, verbose)}
    \begin{Args}[5em]
    	\Item[t]{toolset}{%
		Visual Studio toolset ID\\
		(例) Visual Studio 2015ならば、
		'14.0', '14', 'v140', '2015'の何れでも可。}
    \end{Args}
\end{Initializer}

\begin{Methods}{公開メソッド}
\begin{Method}{solution(name)}
    \begin{Args}[r][5em]
	\Item[t]{name}{ソリューション名}
    \end{Args}
    \begin{Brief}[r]
	引数\tt{name}とVisual Studioのバージョンから
	ソリューションファイル名を決定する。
    \end{Brief}
\end{Method}
\begin{Method}{set(func, arg1, arg2)}
    \begin{Brief}[r]
	与えられた引数に従ってビルド条件の設定を行なう。
	\Vskip{-.5\baselineskip}
	\begin{Table}[r][4em]{|l|l|}\hline
	  \MC{1}{|c}{\it{func}}
	  	& \MC{1}{|c|}{設 定 内 容 (\it{arg1,arg2})}\\\hline
	  \Item[t]{OUTDIR}{%
		出力ディレクトリを\tt{arg1}に、
		クリーンフラグを\tt{arg2}に設定する。
		クリーンフラグがTrueならば、コンパイル前に
		出力ディレクトリをクリアする(強制リコンパイル)。}\hline
	  \Item[t]{LOGFILE}{ログファイルパスを\tt{arg1}とする。}\hline
	  \Item[t]{DRY_RUN}{%
		Dry_runフラグを\tt{arg1}とする。
		このフラグがTrueならば、
		コマンドを表示するだけで実行はしない。}\hline
	\end{Table}
	\Vskip{-.8\baselineskip}
    \end{Brief}
\end{Method}
\begin{Method}{stat = build(platform, config)}
    \begin{Args}[r][5em]
	\Item[t]{platform}{\tt{x86} or \tt{x64}}
	\Item[t]{config}{\tt{Debug}, \tt{Release} or \tt{Trace}}
    \end{Args}
    \begin{Brief}[r]
	ログディレクトリを準備したら、
	プラットフォームを\tt{platform}、ビルド構成を\tt{config}として
	ビルドを実行する(\tt{__build()}を呼ぶ)。
    \end{Brief}
\end{Method}
\begin{Method}{get(kind)}
    \begin{Brief}[r]
	与えられた引数に対応する値を返す。
	\Vskip{-.5\baselineskip}
	\begin{Table}[r][4em]{|l|l|}\hline
	  \MC{1}{|c}{\it{kind}} & \MC{1}{|c|}{返 す 値}\\\hline
	  \Item[t]{VERSION}{Visual Studioのバージョン (e.g. 14.0)}\hline
	  \Item[t]{TOOLSET}{Visual Studio toolset ID (e.g. "v140")}\hline
	  \Item[t]{COMMAND}{
		Visual Studioのtoolset IDと名称\ 
		(e.g. "v140" と "Visual Studio 2015")}\hline
	\end{Table}
	\Vskip{-.8\baselineskip}
    \end{Brief}
\end{Method}
\begin{Method}{info()}
    \begin{Brief}[r]
	現在設定されているパラメータを標準出力に出力する。
    \end{Brief}
\end{Method}
\begin{Method}{\it{string} error()}
    \begin{Brief}[r]
	既にエラーが記録されていたらその情報(文字列)を返す。
    \end{Brief}
\end{Method}
\end{Methods}

\bigskip
\begin{Methods}{内部メソッド \rm{(抜粋)}}
\begin{Method}{\it{list} __get_vsinfo(toolset)}
    \begin{Brief}[r]
	引数で指定されたツールセットに対応する3つ組(pts, vsv, vsn)を返す。
	ここで、ptsはtool set ID (e.g. v140)、vsvはVisual Studioバージョン
	(e.g. 14.0)、vsnはVisual Studio名称 (e.g. Visual Studio 2015) を
	それぞれ表す。\\
	引数で指定されたツールセットが不正なときは(None, None, None)を返す。
    \end{Brief}
\end{Method}
\begin{Method}{\it{string} __get_vs_path(version)}
    \begin{Brief}[r]
	引数で指定されたVisual Studioバージョンに対応するdevenvのパスを返す。
	このメソッドはInitializerから呼ばれる。
	\Vskip{-.5\baselineskip}
	\small{\begin{Table}[r][1em]{ll}
	    \Item{\rm{(注)}}{%
		ここで返すパスは、Visual Studio 2015までについては正しいが、
		それ以降のバージョンについては確認していない。
		必要ならば追加修正する必要がある。}
	\end{Table}}
	\Vskip{-\baselineskip}
    \end{Brief}
\end{Method}
\begin{Method}{\it{int} __build()}
    \begin{Brief}[r]
	devenvに適切なパラメータを与えて実行する。
	クリーンフラグが設定されている場合は、devenvの実行に先立ち、
	出力ディレクトリとこのパラメータに対応するログファイルを削除する。
	devenvのリターンコードを返す。
    \end{Brief}
\end{Method}
\end{Methods}
