\subsubsection{dailybuild\_lib.pm}
\label{subsubsec:dailybuildlib}

\medskip
\noindent
Daily Buildの結果解析で共通に使用する関数のライブラリ

\medskip
\begin{Methods}{サブルーチン}
\begin{Method}{\Cell{%
	read_log((\BS \%modules, \BS @modules, \BS @lines)\\
	\Hskip{60pt}= read_log(\$logfile, \$script [, \$patt_id])}}
    \begin{Args}[r]
	\Item[t]{\$logfile}{ログファイル名}
	\Item[t]{\$script}{呼び出し元スクリプト名(エラー表示用)}
	\Item[t]{\$patt_id}{モジュール開始パターンID(下表参照)}
    \end{Args}
    \vspace{.7\baselineskip}
    \begin{narrow}[10pt]
	\begin{tabular}{cll}\hline
	     id & \MC{1}{c}{pattern} & \MC{1}{c}{用 途}\\\hline
	     0  & \DQuote{*** \it{proj}: \it{solution} ***}
		& \Path{Build.log}及び\Path{Run.log}用\\
	     1  & \DQuote{\it{num}\tt{>} *** \it{proj}: \it{solution} ***}
		& \Path{StubBuild.log}用\\
	     2  & \DQuote{[ \it{alnum} ]}
	    	& \tt{filter.pl}用\\\hline
	\end{tabular}
    \end{narrow}
    \vspace{.7\baselineskip}
    \begin{Return}[r]
	\Item[t]{\%modules}{モジュール名とスレッド情報を関連付けた連想配列。
		キーはモジュール名、値は\tt{\%threads}の参照。
		ここで\tt{\%threads}はスレッド情報を格納する連想配列で、
		キーはスレッド番号、値はそのスレッドに属するログ行。}
	\Item[t]{@modules}{モジュール名を出現順に格納した配列}
	\Item[t]{@lines}{ログの各行を出現順に格納した配列}
    \end{Return}
    \begin{Brief}[r]
	指定されたログファイルを読み込み、モジュール毎にスレッド順に整理する。\\
	モジュールの開始パターンは3種類あり、どれを使うかは引数で指定する
	(引数の説明参照)。\\
	スレッド関連より前に出力されたログ行は、
	スレッド番号$-1$のスレッドに属するものとして扱う。
	また、スレッド関連より後に出力されたログ行は、
	スレッド番号$999$のスレッドに属するものとして扱う。
    \end{Brief}
\end{Method}
\begin{Method}{\Cell{%
	\$number = outside_threads_before()\\
	\$number = outsize_threads_after()}}
    \begin{Brief}[r]
	それぞれ、スレッド情報より前の部分に対する仮想スレッド番号($-1$)
	及びスレッド情報より後の部分に対する仮想スレッド番号($999$)を返す。
    \end{Brief}
\end{Method}
\begin{Method}{\Cell{%
	(\BS @modules1, \BS @modules2)\\
	\Hskip{60pt}= associate_modules(\BS @modules1, \BS@modules2, \$script)}}
    \begin{Args}[r]
	\Item[t]{\BS @modules1}{比較の元(``左側'')となる入力行の配列}
	\Item[t]{\BS @modules2}{比較する(``右側'')側の入力行の配列}
    \end{Args}
    \begin{Brief}[r]
	引数で指定された2つのモジュール配列(\tt{@modules1}, \tt{@modules2}とする)の
	対応する要素が同じモジュールとなるように各配列の要素を調整する。
	そのために、
	\tt{@modules1}にしかないモジュールに対しては
	\tt{@modules2}に削除モジュールマーク(`\tt{_DEL_}')を挿入することで、
	\tt{@modules2}にしかないモジュールに対しては
	\tt{@modules1}に挿入モジュールマーク(`\tt{_INS_}')を挿入することで対応をとる。
	挿入モジュールマーク及び削除モジュールマークは
	\tt{mod_ins()}及び\tt{mod_del()}で取得する。下図参照。\\
	\begin{narrow}[\WID]
	\def\CLINE{\cline{1-1}\cline{3-3}\cline{5-5}\cline{7-7}}
	%\def\RA{ \Hskip{10pt}$\Rightarrow$\Hskip{10pt} }
	\begin{tabular}{|c|c|c|c|c|c|c|}
		\MC{3}{c}{input} & \MC{1}{c}{} & \MC{3}{c}{output}\\\CLINE
		1 && 1 & \RArrowSp{10pt} & 1 && 1\\\CLINE
		2 && X & \RArrowSp{10pt} & \tt{_ins_} && X\\\CLINE
		3 && 2 & \RArrowSp{10pt} & 2 && 2\\\CLINE
		4 && 3 & \RArrowSp{10pt} & 3 && 3\\\CLINE
		5 && 5 & \RArrowSp{10pt} & 4 && \tt{_del_}\\\CLINE
		\MC{1}{c}{modules1} & \MC{1}{c}{} & \MC{1}{c}{modules2} & \MC{1}{c|}{}
			& 5 && 5\\\cline{5-5}\cline{7-7}
		\MC{4}{c}{} & \MC{1}{c}{\tt{modules1}} &
		\MC{1}{c}{} & \MC{1}{c}{\tt{modules2}}
	\end{tabular}
	\end{narrow}
    \end{Brief}
\end{Method}
\begin{Method}{\Cell{%
	mod_ins()\\
	mod_del()}}
    \begin{Brief}[r]
	それぞれ、挿入モジュールマーク(\tt{`_INS_'})、
	削除モジュールマーク(\tt{`_DEL_'})を返す。
    \end{Brief}
\end{Method}
\begin{Method}{\BS @buff = gather_lines(\BS \%modules, \$module)}
    \begin{Args}[r]
	\Item[t]{\%modules}{モジュール名と入力行を対応付けた連想配列}
	\Item[t]{\$module}{抽出するモジュールの名前}
    \end{Args}
    \begin{Return}[r]
	\Item[t]{@buff}{\tt{\%modules}から該当するモジュールの入力行を抽出した配列}
    \end{Return}
    \begin{Brief}[r]
	モジュール名とスレッド情報を関連付けた連想配列から、
	指定されたスレッドのログのみを抽出する。
	\tt{rad_log()}で返された\tt{\%modules}を処理の対象とする。
    \end{Brief}
\end{Method}
\end{Methods}
