\subsubsection{クラス Traverse}
\label{subsubsec:Traverse}

\medskip
\noindent
\bf{概要}

\begin{narrow}[20pt]
指定されたディレクトリを起点としてディレクトリを走査し、
テスト制御ファイルの内容に従ってビルド＆実行を制御する。
テスト結果はイニシャライザに与えられた\ARG{result}に設定される。
\end{narrow}

\bigskip
\noindent
\bf{イニシャライザ (\tt{__init__})}

\vspace{5pt}
\begin{narrow}[20pt]
    \framebox[\linewidth][l]{%
	\begin{tabular}{l}
	    \tt{obj = Traverse(testid, result, csc, control, section,}\\
	    \tt{\hspace{70pt}toolset, platforms, configs, csusage, rebuild,}\\
	    \tt{\hspace{70pt}timeout, report=True, audit=False,}\\
	    \tt{\hspace{70pt}dry_run=False, verbose=0)}
	\end{tabular}}

    \medskip
    引数
    \vspace{-5pt}
    \begin{narrow}[20pt]
	\def\COLWID{300pt}
	\begin{longtable}{ll}
	    \tt{testid}	& \begin{parcol}{\COLWID}{%
		テスト種別(テスト結果ファイルの識別情報として使用する)。\\
		\tt{TESTID.STUB, TESTID.TESTS, TESTID.SAMPLES}のいずれか$^\dagger$
	    }\end{parcol}\\
	    \tt{result}	& \begin{parcol}{\COLWID}{%
		\tt{TestResult}クラスのオブジェクト。テスト結果をここに返す。
	    }\end{parcol}\\
	    \tt{csc}	& \begin{parcol}{\COLWID}{%
	    	\tt{CloseSrcControl}クラスのオブジェクト。
		非公開ソースを使用するか否かを指定する。
	    }\end{parcol}\\
	    \tt{control}& \begin{parcol}{\COLWID}{%
		テスト制御ファイル名。
		DailyBuildでは\FILE{dailybuild.control}
	    }\end{parcol}\\
	    \tt{section}& \begin{parcol}{\COLWID}{%
		テスト制御ファイルのセクション名。
		Windowsのときは\tt{Windows}、unixのときは\tt{unix}を指定する。
	    }\end{parcol}\\
	    \tt{toolset}& \begin{parcol}{\COLWID}{%
		使用するCコンパイラのバージョン。
		WindowsのときはVisual Studioのバージョン、unixのときはダミー。
	    }\end{parcol}\\
	    \tt{platform} & \begin{parcol}{\COLWID}{%
		テストで使用するプラットフォーム。定数\TT{PLATS}のいずれか$^\dagger$
	    }\end{parcol}\\
	    \tt{configs}& \begin{parcol}{\COLWID}{%
		テストで使用するビルド構成。定数\TT{CONFS}のいずれか$^\dagger$
	    }\end{parcol}\\
	    \tt{cuusage}& \begin{parcol}{\COLWID}{%
		非公開ソースを使用するか否か。定数\TT{CSU}のいずれか$^\dagger$
	    }\end{parcol}\\
	    \tt{rebuild}& \begin{parcol}{\COLWID}{%
		強制的に再コンパイルするか否か(論理値で指定)。
	    }\end{parcol}\\
	    \tt{timeout}& \begin{parcol}{\COLWID}{%
		テスト実行時のタイムアウト値(秒で指定)。
		タイムアウトさせないときは値\TT{0}を指定する。
	    }\end{parcol}\\
	    \tt{report}	& \begin{parcol}{\COLWID}{%
		テスト実行状況レポート指定(論理値で指定)。
	    }\end{parcol}\\
	    \tt{dry_run} & \begin{parcol}{\COLWID}{%
		コマンドを表示するだけで実行はしない(論理値で指定)。
	    }\end{parcol}\\\hline
	\end{longtable}
    \end{narrow}

    \vspace{-25pt}
    \small{\begin{tabular}{rl}
	$^\dagger$ & \REF{subsubsec}{ConstDefs}
    \end{tabular}}
\end{narrow}

\bigskip
\noindent
\bf{公開メソッド}

\vspace{5pt}
\def\COLWID{280pt}
\begin{enumerate}
  \item	\tt{traverse(top)}\\
	引数
	\begin{narrow}[20pt]
	    \begin{tabular}{p{30pt}l}
		\tt{top}
		& \begin{parcol}{\COLWID}{走査開始ディレクトリ}\end{parcol}\\
		戻り値 & リターンコード\\
	    \end{tabular}
	\end{narrow}
	\medskip

	概要
	\begin{narrow}[20pt]
		引数で与えられたディレクトリを起点としてディレクトリツリーを走査し、
		対象となるすべてのディレクトリに対するテストを完了させる。
		個々のディレクトリの処理は\METHOD{process()}で行なう。
		テスト結果は引数で与えられた\ARG{result}に設定される。
	\end{narrow}
	\medskip

	処理の説明
	\begin{narrow}[20pt]
		\begin{enumerate}
		  \item	引数\ARG{top}で指定されたディレクトリに移動する。

	          \item	テスト制御ファイルを読む。
			次の条件がすべて成立したら\METHOD{process()}を呼び出して
			このディレクトリ自身の処理を行なう。
			\begin{itemize}
			  \item	このディレクトリが``対象外ディレクトリ''ではない。
				すなわち、
				\begin{itemize}
				  \item	ディレクトリ名が\TT{'.'}で始まらない。
				  \item	ディレクトリ名が\TT{Template}でも
				  	\TT{log}でもない。
				  \item	ディレクトリ名が\ARG{toolset}で与えられた
					名前と等しくない。
				\end{itemize}
				(\METHOD{__is_candidate_dir()}で判定する)。
			  \item	テスト制御ファイルでExclude指定されていない。
			  \item	このディレクトリにsolution/makeファイルがある
				(\METHOD{__has_solution_file()}で判定する)。
			\end{itemize}

		  \item	テスト制御ファイルでDescendが指定されていたら、
			すべての``対象サブディレクトリ''に対して、
			\METHOD{traverse()}を再帰的に呼び出して処理を行なう。
			ただし途中で\verb|^C|割込があったら処理を中止する。
		\end{enumerate}
	\end{narrow}

  \item	\tt{process(cwd, ctl)}\\
	引数
	\begin{narrow}[20pt]
	    \begin{tabular}{p{30pt}l}
		\tt{cwd}
		& \begin{parcol}{\COLWID}{現在ディレクトリ}\end{parcol}\\
		戻り値 & リターンコード\\
	    \end{tabular}
	\end{narrow}
	\medskip

	概要
	\begin{narrow}[20pt]
		カレントディレクトリ(のみ)に対するテストを実行する。
	\end{narrow}
	\medskip

	処理の説明
	\begin{narrow}[20pt]
		\begin{enumerate}
	          \item	solution/makeファイル名を決定する。
			デフォルトのファイル名は
			\Tt{"}<\iT{dierctory-name}><\it{compiler-version}>\tT{.sln}"%
			であるが、テスト制御ファイルのAlias指定により
			変更することができる
			(\METHOD{__solution_file_name()}で判定する)。

		  \item	イニシャライザの引数\ARG{csc}およびテスト制御ファイルの
			UseClosedSrc指定に従って、非公開ソースの使用/未使用を設定する。
			(\METHOD{set_usage()}を呼ぶ。\REF{subsubsec}{ClosedSrcControl})。

		  \item	\tT{BuildAndRun}クラスのオブジェクト\VAR{bar}を準備する。

		  \item	指定されたプラットフォームとビルド構成のすべての組合せに
			対して、
			\begin{itemize}
			  \item	\METHOD{bar.build()}を呼び出してコンパイルを実行する。
		  	  \item	コンパイルが正常でかつ\,\verb|^C|\,割込が
				なかった場合に限り、\METHOD{bar.run()}を呼び出して
				生成されたバイナリを実行する。
			  \item	コンパイル結果および実行結果は\TT{result}に設定する
				(\METHOD{set_result()}を呼ぶ。
				\REF{subsubsec}{TestResult})。
			\end{itemize}
		\end{enumerate}
	\end{narrow}
\end{enumerate}

\bigskip
\noindent
\bf{内部メソッド \rm{(抜粋)}}

\vspace{5pt}
\def\COLWID{280pt}
\begin{enumerate}
  \item	\tt{__init_log(path, step, errlog)}\\
	概要
	\begin{narrow}[20pt]
		ログファイルをオープンしてヘッダ行を書き込む。
		ヘッダ行は次の3行からなる。
		\begin{itemize}
		  \item	日付行
		  \item	ログの識別行 --- 引数\ARG{step}および\ARG{erlog}を元に作成する
		  \item	空行
		\end{itemize}
	\end{narrow}
	\medskip

  \item	\tt{__chdir(path)}\\
	概要
	\begin{narrow}[20pt]
		\METHOD{os.chdir}のラッパ。指定されたディレクトリに移動する。
		移動先としてカレントディレクトリが指定されたときは何もしない。
	\end{narrow}
	\medskip

  \item	\tt{__is_candidate_dir(dir)}\\
	概要
	\begin{narrow}[20pt]
		指定されたディレクトリがテスト対象ディレクトリかどうか検査する。
		ディレクトリ名が次の何れかに該当するならば
		テスト対象外のディレクトリである
		(これらのディレクトリに\METHOD{traverse()}することを防ぐ)。
		\begin{itemize}
		  \item	ディレクトリ名が\TT{'.'}で始まる。
		  \item	ディレクトリ名が\TT{Template}または\TT{log}である。
		  \item	ディレクトリ名が\ARG{toolset}で与えられた名前と一致する。
		\end{itemize}
	\end{narrow}
	\medskip

  \item	\tt{__solution_file_name(ctl, dir, ccver)}\\
	概要
	\begin{narrow}[20pt]
		与えられた引数を元に、solution/makeファイル名を決定する。
		ファイル名はディレクトリ名(\tt{dir})と同じとするが、
		テスト制御パラメータ\VAR{CFK.SOLUTION_ALIAS}
		(\REF{subsubsec}{ConstDefs})で変更することができる。
		Windowsの場合には、末尾にVisual Studioバージョン(\ARG{ccver})および
		拡張子\TT{.sln}を付加する。
	\end{narrow}
	\medskip

  \item	\tt{__has_solution_file(ctl, dir, ccver)}\\
	概要
	\begin{narrow}[20pt]
		カレントディレクトリにソリューションファイルが存在するか否かを
		判定する。ファイルが存在しなければテストの対象外となる。
	\end{narrow}
	\medskip

  \item	\tt{__make_outpath(ctl, slnfile)}\\
	概要
	\begin{narrow}[20pt]
		出力バイナリのパス名を決定する。\\
		出力ディレクトリは、テスト制御パラメー\VAR{CFK.OUTPUT_DIR}が
		指定されていればそこに、指定がなければカレントディレクトリとする。\\
		ファイル名は、テスト制御パラメータ\VAR{CFK.BINARY_OUT}が
		指定されていればそれに、指定がなければソリューションファイル名と
		同じ(拡張子\TT{.sln}は除く)とする。

		\vspace{5pt}
		\footnotesize{%
		※ テスト制御パラメータ\VAR{CFK.OUTPUT_DIR}に指定されたマクロは、
		ここで次の値に変換される。\\
		\begin{tabular}{ll}
		    \MACRO{TOOLSET}
			& イニシャライザの引数\ARG{toolset}の値\\
		    \MACRO{PLATFORM}
			& \METHOD{process()}で設定された変数\VAR{self.platform}の値\\
		    \MACRO{CONFIGURATION}
			& \METHOD{process()}で設定された変数\VAR{self.config}の値
		\end{tabular}}
	\end{narrow}
	\medskip

  \item	\tt{__runtime_addpath(ctl, platform)}\\
	概要
	\begin{narrow}[20pt]
		次のパスリストを返す。これらは、 バイナリ実行時に加えるべきパスである。
		\begin{itemize}
		  \item	テスト制御パラメータ\VAR{CFK.ADD_PATH}に指定されている値。
		  \item	Windowsの場合、
			\PATH{.../dependency/bin/Win64;.../dependency/bin/Win32}
			または
			\PATH{.../dependency/bin/Win32}
			(プラットフォーム\ARG{platform}に従う）。
		\end{itemize}
	\end{narrow}
	\medskip

  \item	\tt{__report(msg_l, msg_s, header, newline)}\\
	\tt{__report_1(msg, header, newline)}\\
	概要
	\begin{narrow}[20pt]
		テストの経過報告を標準出力にレポートする。
		レポート出力の本体は\METHOD{__report()}であり、
		その形式は次のとおり。
		\begin{narrow}\relax
			[\IT{dir-name}\tt{:} ]<\IT{msg}>[\IT{nl}]
		\end{narrow}
		\IT{dir-name}と\IT{nl}の出力は、それぞれ引数\ARG{header}、
		\ARG{newline}にTrueが指定された場合のみ出力される。
	\end{narrow}
	\medskip

  \item	\tt{__status_str(step, status)}\\
	概要
	\begin{narrow}[20pt]
		引数\ARG{status}(数値)に対応した適切な文字列を返す。
	\end{narrow}

\end{enumerate}
